  set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -pthread")

add_library(
  gtest
  gtest/src/gtest-all.cc
  )

macro(add_cods_test name)
  add_executable(
    ${name}
    ${ARGV}
    )
  add_test(
    NAME ${name}
    COMMAND ${name}
    )
  target_link_libraries(
    ${name}
    gtest
    )
  include_directories(
    gtest/include
    ..
    )
endmacro(add_cods_test)

add_cods_test(
  Vector
  Vector.cpp
  )

  HashMap
  HashMap.cpp
  )
