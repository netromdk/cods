408 void HashMap<Key, T, INIT_CAP>::checkRehash() {
409  // If next item fits then stop.
410  if (items + 1 <= buckets.capacity()) {
411  return;
412  }

428 typename HashMap<Key, T, INIT_CAP>::Iterator
429 HashMap<Key, T, INIT_CAP>::_insert(const Key &key, const T &value, bool multi) {
430  // If there is a collision on a key then either override or prepend if multi mode. Otherwise, if
431  // the keys are different then make room for the new one and rehash.
432  for (;;) {
433  auto idx = hashIndex(key);
434  auto *bucket = buckets[idx];
435  if (!bucket) break;
436 
437  if (bucket->key() == key) {
438  if (!multi) {
439  bucket->setValue(value);
440  }
441  else {
442  bucket->addValue(value);
443  items++;
444  }
445  return createIterator(idx);
446  }
447 
448  buckets.append(nullptr);
449  }
450 
451  checkRehash();
452  auto idx = hashIndex(key);
453  buckets[idx] = new Bucket(key, value);
454  items++;
455  return createIterator(idx);
456 }
